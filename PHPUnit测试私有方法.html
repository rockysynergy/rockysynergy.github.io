PHPUnit测试私有方法

更新于：2019-02-18 12:44:32 PHP

测试private方法的代码


    protected static function getMethod($name) {
        $class = new \ReflectionClass('MyClass');
        $method = $class->getMethod($name);
        $method->setAccessible(true);
        return $method;
    }

    public function testFoo() {
        $foo = self::getMethod('foo');
        $obj = new MyClass();
        $foo->invokeArgs($obj, []);
    }
