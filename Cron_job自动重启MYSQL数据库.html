Cron job自动重启MYSQL数据库

更新于：2019-08-10 15:19:48 MySql Shell 

#!/bin/bash
if [[ ! "$(/usr/sbin/service mysql status)" =~ "start/running" ]]
then
    /usr/sbin/service mysql start
fi
  1. 添加到cron job
    chmod +x /path/to/script
sudo crontab -e
*/1 * * * * /path/to/script
